// 1. 基礎設定與資料初始化 const TARGET_DATE = new Date('2026-05-14T07:05:00'); let expenses = JSON.parse(localStorage.getItem('kansai_expenses') || '[]'); let todos = JSON.parse(localStorage.getItem('kansai_todos') || '[]'); // 2. 頁籤切換邏輯 (TAB Switch) function sw(tabId, btn) { // 隱藏所有 tab document.querySelectorAll('.tab').forEach(t => t.classList.remove('on')); // 顯示目標 tab const target = document.getElementById('t-' + tabId); if (target) target.classList.add('on'); // 處理導覽列按鈕狀態 document.querySelectorAll('.bnav button').forEach(b => b.classList.remove('on')); btn.classList.add('on'); // 只有在記帳頁面顯示 FAB 浮動按鈕 document.getElementById('fabBtn').style.display = (tabId === 'expense') ? 'flex' : 'none'; } // 預訂子頁籤切換 function bTab(subId, btn) { const sections = ['flight', 'hotel', 'ticket', 'coupon', 'emergency']; sections.forEach(id => { const el = document.getElementById('b-' + id); if(el) el.style.display = 'none'; }); const targetSub = document.getElementById('b-' + subId); if (targetSub) targetSub.style.display = 'block'; document.querySelectorAll('.tg .tgb').forEach(b => b.classList.remove('on')); btn.classList.add('on'); } // 3. 行程渲染功能 (解決 Day 4 之後無法顯示的問題) function renderSchedule() { const timeline = document.getElementById('timeline'); const dateScroll = document.getElementById('dateScroll'); if(!timeline) return; // 渲染日期捲軸 dateScroll.innerHTML = DAYS.map((day, i) => `
${day.date}
${day.dow}
`).join(''); showDay(0, null); // 預設顯示第一天 } function showDay(idx, btn) { if(btn) { document.querySelectorAll('.dp').forEach(d => d.classList.remove('on')); btn.classList.add('on'); } const day = DAYS[idx]; const tl = document.getElementById('timeline'); tl.innerHTML = day.items.map(item => `
${item.time}${item.est?'
(預)':''}
${item.type.toUpperCase()}
${item.name}
${item.hours || ''}
${item.note}
`).join(''); } function getBadgeColor(type) { const colors = {sight:'#3A6A1A', food:'#8A3A0A', transport:'#0A4A2A', hotel:'#5A1A7A', shopping:'#7A4A0A'}; return colors[type] || '#888'; } // 4. 地址彈窗功能 function openAddr(name, addr) { if(!addr) return; document.getElementById('addrName').innerText = name; document.getElementById('addrText').innerText = addr; document.getElementById('addrFull').classList.add('on'); } function closeAddr() { document.getElementById('addrFull').classList.remove('on'); } function navigateFromAddr() { const addr = document.getElementById('addrText').innerText; window.open(`https://www.google.com/maps/search/?api=1&query=${encodeURIComponent(addr)}`); } // 5. 倒數計時與初始化 function updateCountdown() { const now = new Date(); const diff = Math.ceil((TARGET_DATE - now) / (1000 * 60 * 60 * 24)); document.getElementById('cdnum').innerText = diff > 0 ? diff : 0; } // 6. 記帳功能 (LocalStorage 版本) function addExp() { const name = document.getElementById('en').value; const amount = parseInt(document.getElementById('ea').value); const category = document.getElementById('ecat').value; if(!name || !amount) return alert('請輸入名稱與金額'); expenses.push({ name, amount, category, time: new Date().toLocaleTimeString() }); localStorage.setItem('kansai_expenses', JSON.stringify(expenses)); updateExpenseUI(); toggleExpForm(); // 清空輸入 document.getElementById('en').value = ''; document.getElementById('ea').value = ''; } function updateExpenseUI() { const list = document.getElementById('expList'); const totalJPY = expenses.reduce((s, e) => s + e.amount, 0); const rate = parseFloat(document.getElementById('rateInput').value) || 0.22; document.getElementById('s-jpy').innerText = `¥${totalJPY.toLocaleString()}`; document.getElementById('s-twd').innerText = `NT$${Math.round(totalJPY * rate).toLocaleString()}`; list.innerHTML = expenses.map((e, i) => `
${e.category.split(' ')[0]}
${e.name}
${e.time}
¥${e.amount}
`).reverse().join(''); } function toggleExpForm() { const f = document.getElementById('expForm'); f.style.display = (f.style.display === 'none') ? 'block' : 'none'; } // 啟動 window.onload = () => { updateCountdown(); renderSchedule(); updateExpenseUI(); document.getElementById('sync-bar').innerText = "🌸 已切換至本地儲存模式 (GitHub Pages 專用)"; };
