// 1. 基礎設定與資料初始化 const TARGET_DATE = new Date('2026-05-14T07:05:00'); // 確保引用到 index.html 中定義的 DAYS 變數 const _DAYS = typeof DAYS !== 'undefined' ? DAYS : []; let expenses = JSON.parse(localStorage.getItem('kansai_expenses') || '[]'); let todos = JSON.parse(localStorage.getItem('kansai_todos') || '[]'); // 2. 頁籤切換邏輯 (TAB Switch) function sw(tabId, btn) { document.querySelectorAll('.tab').forEach(t => t.classList.remove('on')); const target = document.getElementById('t-' + tabId); if (target) target.classList.add('on'); document.querySelectorAll('.bnav button').forEach(b => b.classList.remove('on')); if (btn) btn.classList.add('on'); // FAB 浮動按鈕僅在記帳頁顯示 const fab = document.getElementById('fabBtn'); if (fab) fab.style.display = (tabId === 'expense') ? 'flex' : 'none'; } // 預訂子頁籤切換 function bTab(subId, btn) { const sections = ['flight', 'hotel', 'ticket', 'coupon', 'emergency']; sections.forEach(id => { const el = document.getElementById('b-' + id); if(el) el.style.display = 'none'; }); const targetSub = document.getElementById('b-' + subId); if (targetSub) targetSub.style.display = 'block'; document.querySelectorAll('.tg .tgb').forEach(b => b.classList.remove('on')); if (btn) btn.classList.add('on'); } // 3. 行程渲染功能 function renderSchedule() { const timeline = document.getElementById('timeline'); const dateScroll = document.getElementById('dateScroll'); if(!timeline || !dateScroll || _DAYS.length === 0) return; dateScroll.innerHTML = _DAYS.map((day, i) => `
${day.date}
${day.dow}
`).join(''); // 修正：初始載入時傳入第一個按鈕 DOM showDay(0, document.getElementById('dp-btn-0')); } function showDay(idx, btn) { const day = _DAYS[idx]; if(!day) return; if(btn) { document.querySelectorAll('.dp').forEach(d => d.classList.remove('on')); btn.classList.add('on'); } const tl = document.getElementById('timeline'); tl.innerHTML = day.items.map(item => `
${item.time}${item.est?'
(預)':''}
${item.type.toUpperCase()}
${item.name}
${item.hours || ''}
${item.note}
`).join(''); } function getBadgeColor(type) { const colors = {sight:'#3A6A1A', food:'#8A3A0A', transport:'#0A4A2A', hotel:'#5A1A7A', shopping:'#7A4A0A'}; return colors[type] || '#888'; } // 4. 地址彈窗與導航功能 (修復重點) function openAddr(name, addr) { if(!addr) return; document.getElementById('addrName').innerText = name; document.getElementById('addrText').innerText = addr; document.getElementById('addrFull').classList.add('on'); } function closeAddr() { document.getElementById('addrFull').classList.remove('on'); } function navigateFromAddr() { const addr = document.getElementById('addrText').innerText; if (!addr) return; // 修復：正確的 Google Maps 搜尋網址 const url = `https://www.google.com/maps/search/?api=1&query=${encodeURIComponent(addr)}`; window.open(url, '_blank'); } // 5. 倒數計時 function updateCountdown() { const now = new Date(); const diff = Math.ceil((TARGET_DATE - now) / (1000 * 60 * 60 * 24)); const cdNum = document.getElementById('cdnum'); if (cdNum) cdNum.innerText = diff > 0 ? diff : 0; } // 6. 記帳功能 (修復幣別與換算邏輯) function addExp() { const name = document.getElementById('en').value; const rawAmount = parseFloat(document.getElementById('ea').value); const currency = document.getElementById('ec').value; // 獲取幣別 const category = document.getElementById('ecat').value; const rate = parseFloat(document.getElementById('rateInput').value) || 0.22; if(!name || isNaN(rawAmount)) return alert('請輸入名稱與金額'); // 統一換算為 JPY 儲存 let amountInJPY = rawAmount; if (currency === 'TWD') { amountInJPY = Math.round(rawAmount / rate); } expenses.push({ name, amount: amountInJPY, category, time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) }); localStorage.setItem('kansai_expenses', JSON.stringify(expenses)); updateExpenseUI(); toggleExpForm(); // 清空 document.getElementById('en').value = ''; document.getElementById('ea').value = ''; } function updateExpenseUI() { const list = document.getElementById('expList'); if (!list) return; const totalJPY = expenses.reduce((s, e) => s + e.amount, 0); const rate = parseFloat(document.getElementById('rateInput').value) || 0.22; document.getElementById('s-jpy').innerText = `¥${totalJPY.toLocaleString()}`; document.getElementById('s-twd').innerText = `NT$${Math.round(totalJPY * rate).toLocaleString()}`; // 計算人均 (假設團員 4 人) const pp = Math.round(totalJPY / 4); const ppEl = document.getElementById('s-pp'); if (ppEl) ppEl.innerText = `¥${pp.toLocaleString()}`; list.innerHTML = expenses.slice().reverse().map((e) => `
${e.category.split(' ')[0]}
${e.name}
${e.time}
¥${e.amount.toLocaleString()}
`).join(''); } function toggleExpForm() { const f = document.getElementById('expForm'); f.style.display = (f.style.display === 'none' || f.style.display === '') ? 'block' : 'none'; } // 啟動 window.onload = () => { updateCountdown(); renderSchedule(); updateExpenseUI(); const syncBar = document.getElementById('sync-bar'); if (syncBar) syncBar.innerText = "🌸 已切換至本地儲存模式 (GitHub Pages 專用)"; };
